CREATE TABLE CLIENT (
id_client int PRIMARY KEY,
prenom varchar(30),
nom varchar(30),
numero int,
nbPoint int NULL,
nbHeureGratuite int NULL,
)


CREATE TABLE TYPE_FORFAIT (
id_typeForfait varchar(8) PRIMARY KEY,
prix int,
libelle varchar(75),
nb_heure int,
nb_moisValide int
)


CREATE TABLE FORFAIT (
id_forfait int PRIMARY KEY,
date_FinValidite Date,
nb_heureDisponible int,
fk_client int,
fk_typeForfait varchar(8),
FOREIGN KEY (fk_client) REFERENCES CLIENT(id_client), 
FOREIGN KEY (fk_typeForfait) REFERENCES TYPE_FORFAIT(id_typeForfait)
)


CREATE TABLE SALLE (
id_salle int PRIMARY KEY,
nom varchar(30),
prix1H int,
prix2H int,
type_salle varchar(20)
)


CREATE TABLE RESERVATION(
id_reservation int PRIMARY KEY,
datePriseReservation Date NOT NULL,
dateReservation Date NOT NULL,
etatPaiement boolean,
prix int,
duree int,
fk_client int,
fk_salle int,
FOREIGN KEY (fk_client) REFERENCES CLIENT (id_client),
FOREIGN KEY (fk_salle) REFERENCES SALLE (id_salle)
)

CREATE TABLE PLAGE_HORAIRE (
id_trancheH int PRIMARY KEY,
heure_debut int,
heure_fin int,
duree int,
tranche varchar(5)
)

CREATE TABLE RESER_PLAGEHORAIRE (
fk_reservation int,
fk_plagehoraire int,
PRIMARY KEY(fk_reservation, fk_plagehoraire)
)
